Run this in Supabase → SQL Editor to create the required table: 

CREATE TABLE IF NOT EXISTS tll_schedules (
  id             uuid        DEFAULT gen_random_uuid() PRIMARY KEY,
  title          text        NOT NULL,
  youtube_url    text        NOT NULL,
  start_time     integer     DEFAULT 0,
  end_time       integer     NOT NULL,
  scheduled_date date        NOT NULL,
  streaming_time text        NOT NULL,
  is_played      boolean     DEFAULT false,
  played_at      timestamptz,
  is_active      boolean     DEFAULT true,
  created_at     timestamptz DEFAULT now(),
  updated_at     timestamptz
);

ALTER TABLE tll_schedules ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON tll_schedules FOR ALL USING (true) WITH CHECK (true);
